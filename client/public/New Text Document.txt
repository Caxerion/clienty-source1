<script>
  const qualities = {
    mp4: ['1080p', '720p', '480p', '360p'],
    mp3: ['320 kbps', '192 kbps', '128 kbps']
  };

  const urlInput = document.getElementById('url-input');
  const srcTag = document.getElementById('src-tag');
  const fmtBtns = document.querySelectorAll('.fmt-btn');
  const qualitySelect = document.getElementById('quality-select');
  const processBtn = document.getElementById('process-btn');
  const errorMsg = document.getElementById('error-msg');
  const processing = document.getElementById('processing');
  const stubWrap = document.getElementById('stub-wrap');
  const downloadBtn = document.getElementById('download-btn');
  const demoNote = document.getElementById('demo-note');

  let currentFmt = 'mp4';

  function populateQuality(){
    qualitySelect.innerHTML = qualities[currentFmt]
      .map(q => `<option value="${q}">${q}</option>`)
      .join('');
  }
  populateQuality();

  function detectSource(url){
    const u = url.toLowerCase();
    if(u.includes('youtube.com') || u.includes('youtu.be')) return 'YouTube';
    if(u.includes('tiktok.com')) return 'TikTok';
    if(u.includes('instagram.com')) return 'Instagram';
    if(u.includes('twitter.com') || u.includes('x.com')) return 'X / Twitter';
    if(u.includes('facebook.com') || u.includes('fb.watch')) return 'Facebook';
    if(u.trim() === '') return '';
    return 'Unrecognized';
  }

  urlInput.addEventListener('input', () => {
    const src = detectSource(urlInput.value);
    if(src){
      srcTag.textContent = src;
      srcTag.classList.add('show');
    } else {
      srcTag.classList.remove('show');
    }
    errorMsg.textContent = '';
  });

  fmtBtns.forEach(btn => {
    btn.addEventListener('click', () => {
      fmtBtns.forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      currentFmt = btn.dataset.fmt;
      populateQuality();
    });
  });

  function randomSize(fmt){
    if(fmt === 'mp3'){
      return (Math.random()*6 + 2).toFixed(1) + ' MB';
    }
    return (Math.random()*180 + 20).toFixed(0) + ' MB';
  }

  function randomRuntime(){
    const m = Math.floor(Math.random()*8);
    const s = Math.floor(Math.random()*60).toString().padStart(2,'0');
    return `${m}:${s}`;
  }

  function titleFromUrl(url, source){
    try{
      const u = new URL(url);
      const seg = u.pathname.split('/').filter(Boolean).pop();
      if(seg && seg.length > 3 && !seg.includes('watch')){
        return 'Clip — ' + decodeURIComponent(seg).replace(/[-_]/g,' ').slice(0, 48);
      }
    }catch(e){}
    return source ? `Untitled clip from ${source}` : 'Untitled clip';
  }

  processBtn.addEventListener('click', () => {
    const url = urlInput.value.trim();
    errorMsg.textContent = '';
    stubWrap.classList.remove('show');
    demoNote.classList.remove('show');

    if(!url){
      errorMsg.textContent = 'Enter a URL before submitting a declaration.';
      return;
    }
    let valid = true;
    try{ new URL(url); } catch(e){ valid = false; }
    if(!valid){
      errorMsg.textContent = 'That doesn\'t look like a valid URL.';
      return;
    }

    const source = detectSource(url) || 'Unrecognized';

    processBtn.disabled = true;
    processing.classList.add('show');

    setTimeout(() => {
      processing.classList.remove('show');
      processBtn.disabled = false;

      document.getElementById('stub-source').textContent = 'Source: ' + source;
      document.getElementById('stub-id').textContent = '#' + Math.floor(100000 + Math.random()*899999);
      document.getElementById('stub-title').textContent = titleFromUrl(url, source);
      document.getElementById('stub-format').textContent = currentFmt.toUpperCase();
      document.getElementById('stub-quality').textContent = qualitySelect.value;
      document.getElementById('stub-size').textContent = randomSize(currentFmt);
      document.getElementById('stub-runtime').textContent = randomRuntime();

      stubWrap.classList.add('show');
    }, 1400);
  });

  downloadBtn.addEventListener('click', () => {
    demoNote.classList.add('show');
  });
</script>
